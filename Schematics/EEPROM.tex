\documentclass[tikz,border=2pt]{standalone}
\usepackage[european]{circuitikz}
\usepackage{siunitx}
\usepackage{graphicx}

\usetikzlibrary{backgrounds}

\begin{document}
\begin{circuitikz}[scale=1.5, transform shape, thick, font=\Large]

    % Manual bounding box setting
    \draw[use as bounding box] (-7.5,-1.5) rectangle (5.5,6.0);

    % To see it on github
    \begin{scope}[on background layer]
        \fill[white] (-7.5,-1.5) rectangle (5.5,6.0);
    \end{scope}

    % 24C04 EEPROM
    \node[draw, minimum width=5cm, minimum height=3cm] (EEPROM) at (0,2) {24C04};
    
    % EEPROM pins

    % SDA + pull-up
    \node[right] at ([yshift=1.0cm]EEPROM.west) {SDA};
    \draw ([yshift=1.0cm]EEPROM.west) node[above left, font=\footnotesize]{5} to [short, -*] ++(-1.0,0) to ++(-2.5,0) node[left]{SDA};

    \draw ([xshift=-1.0cm, yshift=1.0cm]EEPROM.west)
    to [R, l_=\raisebox{1.5em}{\rotatebox{-90}{\SI{10}{k\ohm}}}, label distance=5pt] ++(0, 1.5) node[vcc]{+\SI{5}{\volt}};

    \node[right] at (EEPROM.west) {SCL};
    \draw (EEPROM.west) node[above left, font=\footnotesize]{6} to [short, -*] ++(-2.5,0) to ++(-1.0,0) node[left]{SCL};

    % SCL + pull-up
    \draw ([xshift=-2.5cm]EEPROM.west) -- +(0, 1.0)
    to [R, l_=\raisebox{1.5em}{\rotatebox{-90}{\SI{10}{k\ohm}}}, label distance=5pt] ++(0, 2.5) node[vcc]{+\SI{5}{\volt}};
  
    \node[right] at ([yshift=-1.0cm]EEPROM.west) {WP};
    \draw ([yshift=-1.0cm]EEPROM.west) node[above left, font=\footnotesize]{7} to ++(-1,0);

    % Other pins
    \node[above] at (EEPROM.south) {GND};
    \draw (EEPROM.south) node[below right, font=\footnotesize]{4} to ++(0,-0.5) node[ground]{};

    \node[left] at ([yshift=1.0cm]EEPROM.east) {A0};
    \draw ([yshift=1.0cm]EEPROM.east) node[above right, font=\footnotesize]{1} to ++(1,0) node[ground]{};

    \node[left] at (EEPROM.east) {A1};
    \draw (EEPROM.east) node[above right, font=\footnotesize]{2} to ++(1,0) node[ground]{};
  
    \node[left] at ([yshift=-1.0cm]EEPROM.east) {A2};
    \draw ([yshift=-1.0cm]EEPROM.east) node[above right, font=\footnotesize]{3} to ++(1,0) node[ground]{};

    \node[below] at (EEPROM.north) {Vcc};
    \draw (EEPROM.north) node[above right, font=\footnotesize]{8} to ++(0,0.5) node[vcc]{+\SI{5}{\volt}};

\end{circuitikz}
\end{document}
